       start = 0
       end = len(a)-1
       
       while start <= end:
           middle = (start + end)//2

def selectionSort(a):
    for i in range(len(a)-1):
        min_idx=i
        for j in range(i+1,len(a)):
            if a[min_idx] > a[j]:
                min_idx=j
        a[i], a[min_idx]=a[min_idx],a[i]
